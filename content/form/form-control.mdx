import { FormControl, Input, Textarea, Radio, Button } from 'insites-ui'
import DemoBox from '../../components/DemoBox'

The form control is a wrapper for all other components in Forms section. It adds
an accessible label and allows to handle error message.

## Adding label

<DemoBox>
    <FormControl>
        <FormControl.Label htmlFor="displayName">Display name</FormControl.Label>
        <Input id="displayName" placeholder="Bryan Fury" />
    </FormControl>
</DemoBox>

```jsx
import { FormControl, Input } from 'insites-ui'

<FormControl>
    <FormControl.Label htmlFor="displayName">Display name</FormControl.Label>
    <Input id="displayName" placeholder="Bryan Fury" />
</FormControl>
```

## Handling errors

<DemoBox>
    <FormControl>
        <FormControl.Label htmlFor="displayName2">Display name</FormControl.Label>
        <Input id="displayName2" placeholder="Bryan Fury" hasError />
        <FormControl.Error>Given name is too short.</FormControl.Error>
    </FormControl>
</DemoBox>

```jsx
import { FormControl, Input } from 'insites-ui'

<FormControl>
    <FormControl.Label htmlFor="displayName">Display name</FormControl.Label>
    <Input id="displayName" placeholder="Bryan Fury" hasError />
    <FormControl.Error>Given name is too short.</FormControl.Error>
</FormControl>
```
